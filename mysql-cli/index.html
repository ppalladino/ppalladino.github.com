MySQL CLI 2016-11-27

Command Description
mysql mysql -u [USER] -p shell into mysql
mysql> show databases; lists databases
mysql> use [DB NAME]; open a database
mysql> show tables; lists tables of current database
mysql> describe [TABLE]; show table structure
mysql> show index from [TABLE]; show indexes on a table
mysql> pager less -SFX prettify query results
$ mysql -u username -p database_name < file.sql import from file to database
